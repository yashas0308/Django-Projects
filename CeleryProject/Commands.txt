#Install RabbitMQ (Ubuntu Linux 20.04LTS)

# Run Celery
    celery -A NAMEOFINSTANCE worker --loglevel=info

    celery -A CeleryProject worker --loglevel=info

    [Windows OS]
    celery -A CeleryProject worker -l info --pool=solo
